import pandas as pd
from sklearn.feature_extraction.text import TfidfVectorizer
from sklearn.metrics.pairwise import cosine_similarity
import numpy as np
import plotly.graph_objects as go


file_path = 'SIMILARITE_2.xlsx'
excel_data = pd.ExcelFile(file_path)

datanumia_df = pd.read_excel(excel_data, sheet_name='DATANUMIA')
edf_df = pd.read_excel(excel_data, sheet_name='EDF')

addresses_datanumia = datanumia_df['adresse_datanumia'].tolist()
addresses_edf = edf_df['adresse_edf'].tolist()

# TF-IDF Vectorization
vectorizer = TfidfVectorizer().fit(addresses_datanumia + addresses_edf)
tfidf_datanumia = vectorizer.transform(addresses_datanumia)
tfidf_edf = vectorizer.transform(addresses_edf)

# Cosine similarity
similarity_matrix = cosine_similarity(tfidf_edf, tfidf_datanumia)


best_matches = np.argmax(similarity_matrix, axis=1)
best_scores = np.max(similarity_matrix, axis=1)


for index, score in enumerate(best_scores):
    if score >= 0.8:  # Similarity > 80%
        best_match_index = best_matches[index]
        edf_df.at[index, 'adresse_datanumia'] = addresses_datanumia[best_match_index]
        edf_df.at[index, 'siret'] = datanumia_df.at[best_match_index, 'siret']
        edf_df.at[index, 'Score de similarit'] = score
    else:
        edf_df.at[index, 'adresse_datanumia'] = np.nan
        edf_df.at[index, 'siret'] = np.nan
        edf_df.at[index, 'Score de similarit'] = np.nan


edf_df.to_excel('SIMILARITE_updated.xlsx', sheet_name='EDF', index=False)


nan_count = edf_df['Score de similarit'].isna().sum()

print(f'Nombre de NaNs: {nan_count}')


scores = best_scores[best_scores >= 0.8]  # Filter scores >= 80%
mean = np.mean(scores)
std_dev = np.std(scores)

x = np.linspace(mean - 3*std_dev, mean + 3*std_dev, 100)
y = (1 / (std_dev * np.sqrt(2 * np.pi))) * np.exp(-0.5 * ((x - mean) / std_dev)**2)


percentiles = np.percentile(scores, [25, 50, 75])

fig = go.Figure()


fig.add_trace(go.Scatter(x=x, y=y, mode='lines', name='Courbe de Gauss'))

for percentile in percentiles:
    fig.add_trace(go.Scatter(x=[percentile, percentile], y=[0, max(y)], mode='lines', name=f'{percentile}me centile'))

fig.add_annotation(
    x=mean, y=max(y), text=f'Nombre de NaNs: {nan_count}', showarrow=False, yshift=10
)

# Update layout
fig.update_layout(
    title='Courbe de Gauss des scores de similarit',
    xaxis_title='Score de similarit',
    yaxis_title='Densit',
)

fig.show()

import pandas as pd
from sklearn.feature_extraction.text import TfidfVectorizer
from sklearn.metrics.pairwise import cosine_similarity
import numpy as np
import plotly.express as px
import plotly.graph_objects as go
from scipy.stats import norm

file_path = 'SIMILARITE_2.xlsx'
excel_data = pd.ExcelFile(file_path)

datanumia_df = pd.read_excel(excel_data, sheet_name='DATANUMIA')
edf_df = pd.read_excel(excel_data, sheet_name='EDF')

# Tokenisation
def preprocess_address(address):
    # Convertir en minuscule
    address = address.lower()
    # Supprimer les caractres spciaux
    address = ''.join(e for e in address if e.isalnum() or e.isspace())
    # Optional
    tokens = address.split()
    return ' '.join(tokens)

addresses_datanumia = datanumia_df['adresse_datanumia'].apply(preprocess_address).tolist()
addresses_edf = edf_df['adresse_edf'].apply(preprocess_address).tolist()

# Vectorisation TF-IDF
vectorizer = TfidfVectorizer().fit(addresses_datanumia + addresses_edf)
tfidf_datanumia = vectorizer.transform(addresses_datanumia)
tfidf_edf = vectorizer.transform(addresses_edf)

# Calcul de similarit cosine
similarity_matrix = cosine_similarity(tfidf_edf, tfidf_datanumia)


best_matches = np.argmax(similarity_matrix, axis=1)
best_scores = np.max(similarity_matrix, axis=1)


for index, score in enumerate(best_scores):
    if score >= 0.8:  # Similarit suprieure  80%
        best_match_index = best_matches[index]
        edf_df.at[index, 'adresse_datanumia'] = addresses_datanumia[best_match_index]
        edf_df.at[index, 'siret'] = datanumia_df.at[best_match_index, 'siret']
        edf_df.at[index, 'Score de similarit'] = score


edf_df.to_excel('SIMILARITE_updated_2.xlsx', sheet_name='EDF', index=False)


similarity_scores = pd.DataFrame({
    'adresse_edf': edf_df['adresse_edf'],
    'best_match_datanumia': [addresses_datanumia[idx] for idx in best_matches],
    'similarity_score': best_scores
})

similarity_scores.to_excel('similarity_scores.xlsx', index=False)


fig = px.histogram(similarity_scores, x='similarity_score', nbins=50, title='Distribution des Scores de Similarit')
fig.update_layout(xaxis_title='Score de Similarit', yaxis_title='Nombre d\'Adresses')

mean, std = norm.fit(similarity_scores['similarity_score'])
xmin, xmax = min(similarity_scores['similarity_score']), max(similarity_scores['similarity_score'])
x = np.linspace(xmin, xmax, 100)
p = norm.pdf(x, mean, std)

fig.add_trace(go.Scatter(x=x, y=p * len(similarity_scores) * (xmax - xmin) / 50, mode='lines', name='Gaussienne'))


box_fig = px.box(similarity_scores, y='similarity_score', title='Box Plot des Scores de Similarit')
box_fig.update_layout(yaxis_title='Score de Similarit')

scatter_fig = px.scatter(similarity_scores, y='similarity_score', title='Scatter Plot des Scores de Similarit')
scatter_fig.update_layout(yaxis_title='Score de Similarit')


sorted_scores = np.sort(similarity_scores['similarity_score'])
cdf = np.arange(len(sorted_scores)) / float(len(sorted_scores))
cdf_fig = px.line(x=sorted_scores, y=cdf, title='Fonction de Rpartition Cumulative (CDF)')
cdf_fig.update_layout(xaxis_title='Score de Similarit', yaxis_title='CDF')

thresholds = [0.8, 0.85, 0.9, 0.95]
counts_above_thresholds = [(similarity_scores['similarity_score'] >= t).sum() for t in thresholds]
threshold_fig = px.bar(x=thresholds, y=counts_above_thresholds, title='Nombre d\'Adresses au-dessus des Seuils de Similarit')
threshold_fig.update_layout(xaxis_title='Seuil de Similarit', yaxis_title='Nombre d\'Adresses')
